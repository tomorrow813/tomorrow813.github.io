public int[] sortedSquares(int[] nums) {
        int i = 0 ,j = nums.length -1;
        int k = nums.length-1;
        int[] arr = new int[nums.length];
        while(i<=j){
            if(nums[i]+nums[j]<0){
                arr[k--]=nums[i]*nums[i++];
            }else{
                arr[k--]=nums[j]*nums[j--];
            }
        }
        return arr;
    }

class Solution {
    public void sortColors(int[] nums) {
        partition(nums,0,nums.length-1,1);
    }

    public void partition(int[] nums,int l , int r, int pivot){
        int start = l-1;
        int end = r+1;
        int i = 0;
        while(i<end){
            if(nums[i]<pivot){
                swap(nums,++start,i++);
            }else if(nums[i]==pivot){
                i++;
            }else{
                swap(nums,--end,i);
            }
        }
    }

    public void swap(int[] nums , int i , int j ){
        int tmp = nums[i];
        nums[i] = nums[j];
        nums[j] = tmp; 
    }
}

class Solution {
    public int[] subSort(int[] array) {
        if(array.length==0) return new int[]{-1,-1};
        int l = -1;
        int r = -1;
        //ä»å·¦å¾€åæ‰¾rçš„å€¼ 
        int max = array[0];
        for(int i = 1; i < array.length ; i ++ ){
            if(array[i]>=max){
                max = array[i];
            }else{
                r = i;
            }
        }
        if(r==-1) return new int[]{-1,-1};
        //ä»åå¾€å·¦æ‰¾lçš„å€¼ 
        int min = array[array.length-1];
        for(int j = array.length -2;j>=0;j--){
            if(array[j]<=min){
                min=array[j];
            }else{
                l = j;
            }
        }
        return new int[]{l,r};
    } 
}
