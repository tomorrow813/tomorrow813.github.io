    public ListNode oddEvenList(ListNode head) {
        if(head==null) return head;
        ListNode odddummy = new ListNode(-1);
        ListNode evendummy = new ListNode(-1);
        ListNode cur  = head;
        ListNode odd = odddummy;
        ListNode even = evendummy;
        boolean evenflag = true;
        while(cur!=null){
            if(evenflag){
                odd.next = cur;
                odd=odd.next;
                evenflag = false;
            }else{
                even.next = cur;
                even = even.next;
                evenflag = true;
            }
            cur=cur.next;
        }
        even.next=null;
        odd.next= evendummy.next;
        return odddummy.next;
    }
}

class Solution {
    public void reorderList(ListNode head) {
        if(head==null) return ;

        ListNode left = head;
        ListNode mid = findMid(head);

        ListNode right = mid.next;
        mid.next=null;//å¾ˆé‡è¦ åˆ«å¿˜äº†
        right = reverse(right);
        merge(left,right);
    }
    //è¿”å›å·¦ä¸­èŠ‚ç‚¹
    public ListNode findMid(ListNode head){
        ListNode fast = head;
        ListNode slow = head;
        while(fast.next!=null&&fast.next.next!=null){
            fast=fast.next.next;
            slow=slow.next;
        }
        return slow;
    }

    public ListNode reverse(ListNode head){
        if(head==null) return null;

        ListNode pre = null;
        ListNode cur = head;
        while(cur!=null){
            ListNode next = cur.next;
            cur.next= pre;
            pre =cur;
            cur=next;
        }
        return pre;
    }

    public void merge(ListNode left,ListNode right){
        ListNode l1 =left;
        ListNode l2= right;
        while(l1!=null&&l2!=null){
            ListNode l1next = l1.next;
            ListNode l2next = l2.next;
            l1.next = l2;
            l2.next=l1next;
            l1 = l1next;
            l2=l2next;
        }
    }
}

class Solution {
    public ListNode sortList(ListNode head) {
        //å½’å¹¶æ’åº
        if(head==null||head.next==null) return head;

        ListNode slow = head;
        ListNode fast = head;
        ListNode first = head;
        while(fast.next!=null&&fast.next.next!=null){
            fast = fast.next.next;
            slow = slow.next;
        }
        ListNode second = slow.next;
        slow.next=null;

        first = sortList(first);
        second = sortList(second);
        ListNode node = merge(first,second);
        return node;
    }

    public ListNode merge(ListNode first,ListNode second){
        ListNode l1=first;
        ListNode l2 = second;
        ListNode dummy = new ListNode(-1);
        ListNode cur = dummy;
        while(l1!=null&&l2!=null){
            if(l1.val<l2.val){
                cur.next = l1;
                l1=l1.next;
            }else{
                cur.next=l2;
                l2=l2.next;
            }
            cur = cur.next;
        }
        cur.next = (l1==null)?l2:l1;
        return dummy.next;
    }
}
